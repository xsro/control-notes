\subsection{PE 条件}

\begin{frame}
    \frametitle{PE条件：一个充分条件}

    \[
    \begin{cases}
    \dot{e} = -e + \theta w(t) \\
    \dot{\theta} = -e w(t)
    \end{cases}
    \]

    \begin{lemma}
        Lemma 2.4 \cite{Chen2015} Sec 2.3 p 29.
        考虑一个连续可微的函数$g:[0,\infty)\mapsto \mathbb{R}^n$ 和一个有界分段连续的函数$f:[0,\infty)\mapsto \mathbb{R}^n$满足
        \[\lim_{t\to\infty} g^T(t)f(t)=0\]
        当满足以下两个条件的时候：
        \begin{enumerate}
            \item $\lim_{t\to\infty} \dot{g}(t)=0$
            \item $f(t)$ is PE.
        \end{enumerate}
        有 $\lim_{t\to\infty}=0$.
    \end{lemma}
\end{frame}
